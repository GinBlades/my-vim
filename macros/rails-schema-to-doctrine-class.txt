<?php

namespace App\Archive\Entity;

use Doctrine\ORM\Mapping as ORM;

/**
 * @ORM\Table(name="pvls")
  * @ORM\Entity
   */
   class Pvl
   {
     create_table "pvls", :primary_key => "pvl_id", :force => true do |t|
        t.datetime "created_at"
        t.string   "created_by",                   :limit => 30
   }
}
